Step 1 — Capture the map

With the quest map open on the official site, open DevTools (F12) → Console, paste this and press Enter. It fetches the map's scenario notes (objective, special rules, riddles) and downloads the page with them embedded:
(async()=>{let t='';try{t=await(await fetch('ajax-mapas.php')).text();if(t.trim().length<5){const idm=(location.search.match(/\d+/)||[''])[0];t=await(await fetch('ajax-mapas.php',{method:'POST',headers:{'Content-Type':'application/x-www-form-urlencoded'},body:'idm='+idm})).text();}}catch(e){}const s=document.createElement('script');s.id='dun-annotations';s.type='application/json';s.textContent=t;document.body.appendChild(s);document.querySelectorAll('[tipo="marcadorpersonalizado"]').forEach(function(m){var cs=getComputedStyle(m);m.setAttribute('data-mkcolor',cs.backgroundColor);m.setAttribute('data-mknum',(m.getAttribute('txt')||m.textContent||'').trim());m.setAttribute('data-mkshape',/cuadrado/.test(m.className)?'square':'round');});const a=document.createElement('a');a.href=URL.createObjectURL(new Blob([document.documentElement.outerHTML],{type:'text/html'}));a.download='mapaonline_'+(location.search.match(/\d+/)||['map'])[0]+'.html';a.click()})()
